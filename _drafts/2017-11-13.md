# 表结构设计

```sql
CREATE TABLE `food` (
    `id` bigint(20) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键ID',
    `name` varchar(50) NOT NULL DEFAULT '' COMMENT '名字',
    `alias` varchar(50) NOT NULL DEFAULT '' COMMENT '别名',
    `heat` float NOT NULL DEFAULT '0' COMMENT '每百克热量(大卡)',
    `carbohydrate` float NOT NULL DEFAULT '0' COMMENT '每百克碳水化合物(克)',
    `fat` float NOT NULL DEFAULT '0' COMMENT '每百克脂肪(克)',
     `protein` float NOT NULL DEFAULT '0' COMMENT '每百克蛋白质(克)',
    `classify_id` int(11) NOT NULL DEFAULT '0' COMMENT '分类ID,food_classify id',
    `img` varchar(255) NOT NULL DEFAULT '' COMMENT '图片',
    `create_time`  datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
    PRIMARY KEY (`id`)
)ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='食物数据表';
```

```sql
CREATE TABLE `food_classify` (
    `id` bigint(20) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键ID',
    `classify_name` varchar(50) NOT NULL DEFAULT '' COMMENT '名字',
     `create_time`  datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
    PRIMARY KEY (`id`)
)ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='食物分类表';
```

```sql
CREATE TABLE `food_energy_ruler` (
    `id` bigint(20) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键ID',
    `food_id` bigint(20) NOT NULL DEFAULT '0' COMMENT '食物ID,food ID',
    `food_classify_name` varchar(100) NOT NULL DEFAULT '' COMMENT '度量单位',
    `heat` float NOT NULL DEFAULT '0' COMMENT '热量(大卡)',
    `create_time`  datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
    PRIMARY KEY (`id`)
    )ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='食物热量表';
```
